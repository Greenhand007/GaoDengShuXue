\begin{frame}
\frametitle{对弧长的曲线积分}
\begin{remark}
对弧长的曲线积分又称为\bold{第一类曲线积分}．
\end{remark}
\vpause
\begin{remark}
对闭曲线$L$，记$\int_{L}f(x,y)\ds=\bold{\oint_{L}f(x,y)\ds}$．
\end{remark}
\vpause
\begin{remark}
若$L$是分段光滑曲线，$f(x,y)$在$L$上连续，则
$\int_{L}f(x,y)\ds$存在。
\end{remark}
\vpause
\begin{remark}
$\bold{\int_{L}1\ds=\text{Length}(L)}$，即曲线弧$L$的长度．
\end{remark}
\end{frame}

\begin{frame}
\frametitle{对弧长的曲线积分的性质}
\begin{property}[线性和]
设$\alpha$和$\beta$为常数，则
$$\int_{L}(\alpha f+\beta g)\ds=\alpha\int_{L}f\ds+\beta\int_{L}g\ds$$
\end{property}
\vpause
\begin{property}[分段和]
将积分弧段$L$分成$L_1$和$L_2$，则
$$\int_{L}f(x,y)\ds=\int_{L_{1}}f(x,y)\ds+\int_{L_{2}}f(x,y)\ds$$
\end{property}
\vpause
\begin{property}[保号性]
若在$L$上$f(x,y)\geq g(x,y)$，则
$$\int_{L}f(x,y)\ds \geq \int_{L}g(x,y)\ds$$
\end{property}
\end{frame}

\subsection{计算方法}

\begin{frame}%<99>
\frametitle{第一类曲线积分的计算}
\begin{property*}
设平面曲线$L$的参数方程为
$\left\{\begin{array}{l}x=\phi(t)\\y=\psi(t)\end{array}\right.$，
其中$a\le t\le b$，则弧长元素
\[ \ds=\sqrt{\phi^{\prime}(t)^{2}+\psi^{\prime}(t)^{2}}\dt. \]
\end{property*}
\ppause\cdotfill
$$\cbold
\int_{L}f(x,y)\ds =\int_{a}^{b}f(\phi(t),\psi(t))\sqrt{\phi'(t)^{2}+\psi'(t)^{2}}\dt
$$
\end{frame}

\begin{sframe}
\frametitle{第一类曲线积分的计算}
\begin{theorem*}
设$f(x,y)$在曲线弧$L$上有定义且连续，$L$的参数方程为
\[
\left\{ \begin{array}{c}
x=\phi(t),\\
y=\psi(t)
\end{array}(a\leq t\leq b),\right.
\]
$\phi'(t)$和$\psi'(t)$的导数连续，且$\phi'(t)^{2}+\psi'(t)^{2}\neq0$，则曲线积分存在，而且
\[
\int_{L}f(x,y)\ds=\int_{a}^{b}f[\phi(t),\psi(t)]\sqrt{\phi'(t)^{2}+\psi'(t)^{2}}\dt
\]
\end{theorem*}
\end{sframe}

\begin{frame}
\frametitle{第一类曲线积分的计算}
特别地，若曲线方程为$\bold{y=\psi(x)}$（$a\le x\le b$），则有
\[
\int_{L}f(x,y)\ds=\int_{a}^{b}f[x,\psi(x)]\sqrt{1+\psi'(x)^2}\dx
\]
\vspace{-1em}\pause\vfill\cdotfill\vfill
若曲线方程为$\bold{x=\phi(y)}$（$a\le y\le b$），则有
\[
\int_{L}f(x,y)\ds=\int_{a}^{b}f[\phi(y),y]\sqrt{\phi'(y)^2+1}\dy
\]
\end{frame}

\begin{frame}
\frametitle{复习与提高}
\begin{puzzle}
计算$\oint_{\Gamma}x^2\ds$,其中$\Gamma$为球面$x^2+y^2+z^2=1$被平面$x+y+z=0$所截的圆周.
\end{puzzle}
\stext{\vpause
\begin{method}\smark
先将圆周的方程化成参数方程,然后利用公式进行计算.
\end{method}
\vpause
\begin{method}\smark\let\displaystyle=\textstyle
由对称性，有$\oint_{\Gamma}x^2\ds=\oint_{\Gamma}y^2\ds=\oint_{\Gamma}z^2\ds$.
所以$\begin{aligned}[t]\oint_{\Gamma}x^{2}\ds&=\frac13\oint_{\Gamma}(x^2+y^2+z^2)\ds\\
&=\frac13\oint_{\Gamma}1\ds=\frac13\cdot2\pi=\frac{2\pi}{3}\end{aligned}$\par
其中由于平面过球心，所以圆周的半径为$1$．
\end{method}}
\end{frame}

\begin{frame}
\frametitle{复习与提高}
\begin{puzzle}
计算$I=\oint_{\Gamma}(x^{2}+y^{2}+z^{2})\ds$,其中$\Gamma$为球面
$x^{2}+y^{2}+z^{2}=\frac{9}{2}$被平面$x+z=1$所截得的曲线. 
\end{puzzle}
\stext{\vpause
\begin{solution}\smark
可以利用参数方程来计算：
\[ \begin{cases}
  x = \frac12 + \sqrt2\cos\theta \\
  y = 2\sin\theta \\
  z = \frac12 - \sqrt2\cos\theta
\end{cases}\]
也可以仿照上题做法,只需用点到平面的距离公式求出$\Gamma$的半径为$2$，从而$I=\frac{9}{2}\cdot 4\pi=18\pi$.
\end{solution}}
\end{frame}
