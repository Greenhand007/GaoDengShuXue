\subsection{空间曲线的参数方程}

\begin{frame}
\frametitle{空间曲线的参数方程}
将曲线 $C$ 上动点的坐标 $x$，$y$，$z$ 表示为参数 $t$ 的函数\pause
\[\left\{\begin{gathered}
  x=x(t)\\
  y=y(t)\\
  z=z(t)
\end{gathered}\right.\]\pause
称它为空间曲线的\bold{参数方程}。
\end{frame}

\begin{frame}
\begin{columns}[onlytextwidth]
\column{.7\textwidth}
\begin{example}
螺旋线
\[\left\{\begin{aligned}
  x&=a\cos\omega t\\
  y&=a\sin\omega t\\
  z&=vt
\end{aligned}\right.\]\pause
\begin{itemize}
  \item 以角速度$\omega$围绕$z$轴旋转\pause
  \item 以线速度$v$平行$z$轴上升
\end{itemize}
\end{example}
\pause
\column{.3\textwidth}
\begin{center}
\begin{tikzpicture}[x={(-2cm,-1cm)},y={(2cm,-1cm)},z={(0cm,3cm)},samples=80,scale=.3,thick]
\drawline{densely dashed}{}{(0,0,0),(1,0,0),(2,0,0)}; \draw[->] (1,0,0) -- (2,0,0) node[left,below] {$x$};
\drawline{densely dashed}{}{(0,0,0),(0,1,0),(0,2,0)}; \draw[->] (0,1,0) -- (0,2,0) node[right] {$y$};
\drawline{densely dashed}{}{(0,0,0),(0,0,4),(0,0,5)}; \draw[->] (0,0,4) -- (0,0,5) node[above] {$z$};
\drawplot[color=accent2]{}{densely dashed}{-.25*pi,.75*pi,1.75*pi}{({cos(\x r)},{sin(\x r)},0)}
\draw[color=accent2] plot[domain=0:2*pi] ({cos(\x r)},{sin(\x r)},4);
\draw[color=accent2] (-.707,.707,0) -- (-.707,.707,4) (.707,-.707,0) -- (.707,-.707,4);
\drawplot[thick,color=accent3]{}{densely dashed}{0,.75*pi,1.75*pi,2.75*pi,3.75*pi,4.75*pi}
                                  {({cos(\x r)},{sin(\x r)},{.268*(\x)})};
\end{tikzpicture}
\end{center}
\end{columns}
\end{frame}

\begin{frame}
\begin{example}
将下列曲线化为参数方程表示：
\[
(1)\left\{\begin{aligned}
x^2+y^2&=1\\
2x+3z&=6
\end{aligned}\right.\pause\quad
(2)\left\{\begin{aligned}
&z=\sqrt{a^2-x^2-y^2}\\
&x^2+y^2-ax=0
\end{aligned}\right.
\]
\end{example}
\end{frame}
