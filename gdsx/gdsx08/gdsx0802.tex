\section{数量积与向量积}

\subsection{两向量的数量积}

\begin{frame}
\frametitle{两向量的数量积}
\begin{example}
设一物体在常力$\vec F$作用下，沿与力夹角为$\theta$的直线移动，位移为$\vec s$。则力$\vec F$所做的功为\pause
\begin{columns}[onlytextwidth]
\column{.6\textwidth}
$$W=|\vec F|\,|\vec s|\cos\theta\onslide<4>{=\bold{\vec F \cdot \vec s}}$$
\column{.4\textwidth}
\begin{tikzpicture}[thick,font=\small,scale=.7]
\draw[->] (0,0) node[left]{$A$} -- node[below,color=accent1]{$\vec{s}$} (3,0) node[right]{$B$};
\draw[->] (0,0) -- node[above,color=accent1]{$\vec{F}$} (2,2);
\draw[draw=accent5] (.4,0) arc (0:45:.4); \node[above,inner sep=2pt,color=accent1] at (.8,0) {$\theta$};
\end{tikzpicture}
\end{columns}
\end{example}
\pause
\begin{definition}
设向量$\vec a$和$\vec b$的夹角为$\theta$，称
\[ \vec a \cdot \vec b = |\vec a|\,|\vec b|\cos\theta \]
为$\vec a$与$\vec b$的\bold{数量积}（点乘）。
\end{definition}
\end{frame}

\begin{frame}
\frametitle{数量积与投影}
\begin{columns}[onlytextwidth]
\column{.73\textwidth}
数量积与投影的关系如下：
\[ \vec a\cdot\vec b = |\vec a|\,\bold{|\vec b|\cos\theta} \pause = \vert\vec a\vert\,\bold{\Prj_{\vec a}\vec b}\]
\column{.27\textwidth}
\pause
\begin{tikzpicture}[thick,font=\small,scale=.7]
\draw[densely dashed] (2,0) -- (2,2);
\draw[->] (0,0) -- (3,0) node[right,color=accent1]{$\vec{a}$};

\draw[draw=accent5] (.4,0) arc (0:45:.4); \node[above,inner sep=2pt,color=accent1] at (.8,0) {$\theta$};
\draw[draw=accent2] (0,0) node[below]{$A$} -- (2,0) node[below]{$B$};
\end{tikzpicture}
\end{columns}
\pause
同理，我们有 $\vec a\cdot\vec b = |\vec b|\,\bold{|\vec a|\cos\theta} = \vert\vec b\vert\,\bold{\Prj_{\vec b}\vec a}$。
\end{frame}

\begin{frame}
\frametitle{数量积的性质}
\begin{property}
规定零向量和任何向量都垂直，则有
$$\bold{\va\cdot\vb=0} \Longleftrightarrow \bold{\va\bot\vb}$$
\end{property}
%\pause
% FIXED：当前面定理以独立公式结尾时，\pause 会导致有多余的空白段落产生，\onslide<2->{} 也一样
% 已解决，是 theorem end 没有用 \par 结尾（见 http://tex.stackexchange.com/q/179575/8956）
% 实际上 $$...$$ 有时也会导致空白问题，最好用 \[...\]（见 http://tex.stackexchange.com/q/503/8956）
\pause
\begin{property}
向量的数量积符合下列运算定律：\pause
\begin{enumerate}[<+->]
  \item $\va\cdot\va=|\va|^2$
  \item $\va\cdot\vb=\vb\cdot\va$
  \item $(\va+\vb)\cdot\vc=\va\cdot\vc+\vb\cdot\vc$
  \item $(\lambda\va)\cdot\vb=\lambda(\va\cdot\vb)$
\end{enumerate}
\end{property}
\end{frame}

\begin{frame}
\frametitle{数量积的坐标表示}
设$\vec a=(a_x,a_y,a_z)$，$\vec b=(b_x,b_y,b_z)$，\pause 则有
$$\vec a \cdot \vec b = a_xb_x+a_yb_y+a_zb_z$$\pause
对于非零向量，由$\vec a \cdot \vec b=|\vec a|\,|\vec b|\cos\theta$，\pause 得
\[ \cos\theta = \frac{\vec a \cdot \vec b}{|\vec a|\,|\vec b|}\pause=\frac{a_xb_x+a_yb_y+a_zb_z}{\pause\sqrt{a_x^2+a_y^2+a_z^2}\sqrt{b_x^2+b_y^2+b_z^2}}\]\pause
此式即为\bold{两向量的夹角公式}。
\end{frame}

\begin{frame}
\frametitle{两向量的向量积}
\begin{definition}
设$\vec a$与$\vec b$的夹角为$\theta$，定义向量$\vec c$\pause
\begin{itemize}
  \item[] 大小：$|\vec c|=|\vec a|\,|\vec b|\,\sin\theta$\pause
  \item[] 方向：$\vec c\bot\vec a$, $\vec c\bot\vec b$ 且 $\vec a$, $\vec b$, $\vec c$ 符合右手定则
\end{itemize}\pause
称$\vec c$为向量$\vec a$与$\vec b$的\bold{向量积}（叉乘），记为$$\bold{\vec a\times \vec b=\vec c}$$
\end{definition}
\end{frame}

\begin{frame}
\frametitle{向量积的性质}
\begin{property}
规定零向量和任何向量都平行，则有
$$\bold{\va\times\vb=\vec0} \Longleftrightarrow \bold{\va\parallel\vb}$$
\end{property}
\pause
\begin{property}
向量的向量积符合下列运算定律：\pause
\begin{enumerate}[<+->]
  \item $\va\times\va=\vec0$
  \item $\vb\times\va=-\va\times\vb$
  \item $(\va+\vb)\times\vc=\va\times\vc+\vb\times\vc$
  \item $(\lambda\va)\times\vb=\lambda(\va\times\vb)$
\end{enumerate}
\end{property}
\end{frame}

\begin{frame}
\frametitle{向量积的坐标表示}
设$\vec a=(a_x,a_y,a_z)$，$\vec b=(b_x,b_y,b_z)$，\pause 则有
\begin{align*}
&\phantom{{}={}}\vec a \times \vec b =\left(
    \begin{vmatrix}a_y&a_z\\b_y&b_z\end{vmatrix},
    \begin{vmatrix}a_z&a_x\\b_z&b_x\end{vmatrix},
    \begin{vmatrix}a_x&a_y\\b_x&b_y\end{vmatrix}\right) \\
&= \Big(a_yb_z-a_zb_y,a_zb_x-a_xb_z,a_xb_y-a_yb_x\Big)
\end{align*}\pause
其中二阶行列式的定义为$\left|\begin{array}{cc}a&b\\c&d\end{array}\right|=ad-bc$。
\end{frame}

\end{choicehalf}
